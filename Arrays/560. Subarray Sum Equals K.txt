560. Subarray Sum Equals K		(HashMap + PrefixSum)

Given an array of integers nums and an integer k, return the total number of subarrays whose sum equals to k.
A subarray is a contiguous non-empty sequence of elements within an array.

Example 1:
Input: nums = [1,1,1], k = 2
Output: 2


</>CODE
import java.util.HashMap;
class Solution {
    public int subarraySum(int[] nums, int k) {
        HashMap<Integer, Integer> map = new HashMap<>();
        map.put(0, 1);      //Initital (K,V)=(sum,count) of HashMap
        int sum = 0;
        int count = 0;

        for (int i = 0; i < nums.length; i++) {     
            sum += nums[i];     //Sum of all the iterated elements

            if (map.containsKey(sum - k)) {     //Check if s-k exists
                count += map.get(sum - k);      //add the count of value(key,value) from HashMap
            }
         
	   map.put(sum, map.getOrDefault(sum, 0) + 1);     //Check if sum already exists in Map, add the frequency +1 and update the freq in the Map.  
        }
        return count;       //return number of subarrays who's sum = k
    }
}
